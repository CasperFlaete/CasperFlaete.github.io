Plotte globalt batymetrykart og indiker dybdetransekt på kartet. ¶

Her har jeg valgt et havbasseng og definert et vest-øst transekt som dekker ulike havdyp og sedimenttyper, inkludert kalsiumkarbonatsedimenter og dyphavsleire. Så plotter jeg transektet mitt som en linje over et kart som viser global batymetri.

#Henter inn modulene vi trenger for å løse denne delen av oppgaven. 
import numpy as np
import pandas as pd
import matplotlib as mpl
import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap

from cpt_convert import loadCPT 
from matplotlib.colors import LinearSegmentedColormap 

import cmocean


#Laster inn filen 'seafloor_lithology_colormap' og konverterer dette til et linært segmentert fargekart.
cpt = loadCPT('seafloor_lithology_colormap.cpt')
cpt_convert = LinearSegmentedColormap('cpt', cpt)

#Laster inn 'oceanographic_data.csv' og kaller denne for bath_data.  
bath_data = pd.read_csv('./data/oceanographic_data.csv')

#Definerer lengdegrader og breddegrader.
n_latitudes = 179 
n_longitudes = 361 

#Lager dimensjonene til kartert som vi skal bruke senere. 
dims = (n_latitudes, n_longitudes)

#Definerer nye variabler og reshaper dimensjonene til lengdegrader, breddegrader, litologi og bathymetri. 
lon = np.reshape(bath_data.longitude.values, dims) 
lat = np.reshape(bath_data.latitude.values, dims)
litho = np.reshape(bath_data.lithology.values, dims)
bath = np.reshape(bath_data.bathymetry.values, dims)


#Definerer figurstørrelse
fig = plt.figure(figsize = (15, 65))

#Definerer aksene til figuren
ax_map = fig.add_axes([0, 0, 0.9, 1.0])

#Lager basemap projektet
m = Basemap(projection = 'robin', lon_0 = 0.0, lat_0 = 0.0, resolution = 'l', ax = ax_map) 


#Setter riktige farger på kartet med lengdegrad, breddegrad, bathymetry. 
mapmesh = m.pcolormesh(lon, lat, bath, latlon = True, cmap = cmocean.cm.deep) 


#Definerer hvor transektet skal lages. 
latitude = [-30,-30] 
longitude = [30,115] 
#Tar verdiene oppgitt i variablene latitude og longitude, deretter plotter disse med basemap modulen.
#Dette transektet vises i rosa farge på kartet.
x, y = m(longitude, latitude)
m.plot(x, y, 'o-', markersize=5, linewidth=5, color ='pink') 

#Fyller inn kontinentene med grå og innsjøfarge hvit. 
m.fillcontinents(color = 'gray', lake_color = 'w')


#Definerer fargepaletten, Sier at vi skal bruke kartfargene til 'cmocean.cm-deep'
cbar = m.colorbar(mapmesh, cmap = cmocean.cm.deep, pad = 0.5)


#Lager lengdegrader og breddegrader på kartet.
m.drawparallels(np.arange(-90., 91., 30.), labels = [True, True, False, False])
m.drawmeridians(np.arange(-180., 181., 45.), labels = [False, False, False, True])

#Setter tittel på kartet.
plt.title('Bathymetry')

#Plotter kartet. 
plt.show()

#Definerer en variabel ved navn 'longitude_transect' som vi setter lik arrayen 'lon', 
#dataene som lagres i denne variabelen er da lengdegrader mellom -30 og 115, på breddegrad -30.
longitude_transect = lon[(lat == -30) & (lon > 30) & (lon < 115)]

#Definerer en variabel ved navn 'bathymetry_transect' som vi setter lik arrayen 'bath', 
#dataene som lagres i denne variabelen er da bathymetriske data mellom -30 og 115 lengdegrad, på breddegrad -30.
bathymetry_transect = bath[(lat == -30) & (lon > 30) & (lon < 115)]

#Definerer en variabel ved navn 'lithology_transect' som vi setter lik arrayen 'litho', 
#dataene som lagres i denne variabelen er da lithologier mellom -30 og 115 lengdegrad, på breddegrad -30.
lithology_transect = litho[(lat == -30) & (lon > 30) & (lon < 115)]


#Definerer figurstørrelse
fig = plt.figure(figsize = (15, 5))

#Definerer aksene til figuren
ax_map = fig.add_axes([0, 0, 0.7, 1.0]) 

#Legger til en tittel på bilde. 
plt.title ("Depthtransect with lithologies") 

#Plotter grafen med longitude_transect på x-aksen og bathymetry_transect på y-aksen. 
plt.plot(longitude_transect, bathymetry_transect, "k-", linewidth = 0.5) 

#Definerer en variabel med navn transect. Denne tar longitude_transect og bathymetry_transect. 
#Deretter setter den riktige farger på punktene på diagrammet utifra lithology_transect. 
transect = plt.scatter(longitude_transect, bathymetry_transect, c = lithology_transect, cmap = cpt_convert, vmin=1, vmax = 13, zorder = 3 )
 
#Setter inn aksetittler.
plt.xlabel("Longitude", labelpad = 20)
plt.ylabel ("Depth", labelpad = 20) 

#Definerer hvordan tegnforklaringen skal være. cbar vil her være et resultat av variabelen transect som vi definerte tidligere i oppgaven. 
bounds = np.linspace(0.5, 13.5, 14)
norm = mpl.colors.BoundaryNorm(bounds, cpt_convert.N)
cbar = plt.colorbar(transect, cmap = cpt_convert, norm = norm,
                spacing = 'proportional', ticks = bounds-.5, boundaries = bounds, 
                format = '%s',location = 'right', pad = 0.1)

#Setter figurforklaring til fargekodene, dette hentes fra listen 'sediment_types'
cbar.set_ticklabels(sediment_types)

#Plotter kartet. 
plt.show()
